\begin{frame}{Input Tensor \& Problem Setup}
\footnotesize

\begin{columns}[T]
\column{0.58\textwidth}

\textbf{Problem:} Build a single model that segments two types of drywall damage ---
taping areas and surface cracks --- conditioned on a text prompt.
Trained from scratch, no pretrained weights.

\medskip
\textbf{Dataset:} 4,646 images (3,638 train / 1,008 val) from Roboflow.
\begin{itemize}
  \item Drywall Taping: 1,022 images (bbox $\to$ filled rect masks)
  \item Surface Cracks: 3,624 images (polygon annotations)
  \item Class imbalance 1:3.4 handled by weighted random sampler
\end{itemize}

\medskip
\textbf{Augmentation (7 transforms):}
horizontal/vertical flip, rotation $\pm$15\textdegree, colour jitter,
Gaussian blur, Gaussian noise, ImageNet normalisation.

\column{0.38\textwidth}

\begin{block}{Image Tensor}
\begin{tabular}{@{}ll@{}}
Shape  & \texttt{(B, 3, 512, 512)} \\
Type   & \texttt{float32} / fp16 \\
Norm   & ImageNet $\mu$, $\sigma$ \\
\end{tabular}
\end{block}

\begin{block}{Text Tensor}
\begin{tabular}{@{}ll@{}}
Shape  & \texttt{(B, 21)} \\
Type   & \texttt{int64} token IDs \\
Vocab  & 128 ASCII chars \\
\end{tabular}
\end{block}

\begin{block}{Output Tensor}
\begin{tabular}{@{}ll@{}}
Shape     & \texttt{(B, 1, 512, 512)} \\
Meaning   & per-pixel logits \\
Threshold & 0.5 $\to$ binary mask \\
\end{tabular}
\end{block}

\end{columns}
\end{frame}

\begin{frame}{Architecture --- Major Blocks}
\footnotesize

\begin{columns}[T]
\column{0.48\textwidth}

\textbf{Vision Encoder} (4-stage DSConv + SE)
\begin{itemize}
  \item Stem: Conv2d(3$\to$32, stride 2)
  \item Stage 1: 2$\times$ DSConv, 32$\to$64
  \item Stage 2: 3$\times$ DSConv, 64$\to$128
  \item Stage 3: 3$\times$ DSConv, 128$\to$256
  \item Stage 4: 2$\times$ DSConv, 256$\to$512
  \item Squeeze-and-Excitation after each stage (reduction=4)
  \item FiLM text conditioning at every stage
\end{itemize}

\medskip
\textbf{ASPP Bottleneck} (multi-scale context)
\begin{itemize}
  \item 5 parallel branches: 1$\times$1 conv $+$ 3$\times$3 atrous at rates 6, 12, 18 $+$ global avg pool
  \item Fusion: concat (640ch) $\to$ 1$\times$1 conv $\to$ 512ch
  \item Dropout2d(0.1)
\end{itemize}

\column{0.48\textwidth}

\textbf{U-Net Decoder} (4 stages + skip connections)
\begin{itemize}
  \item Bilinear 2$\times$ upsample $+$ concat skip $+$ DSConv
  \item FiLM conditioning at 3 decoder stages
  \item Deep supervision aux heads at 1/32 and 1/16
  \item Final: Conv2d(32$\to$1, kernel 1)
\end{itemize}

\medskip
\textbf{Text Encoder} (Char-level BiLSTM)
\begin{itemize}
  \item Embedding: 128 vocab $\to$ 64 dims
  \item BiLSTM: hidden 128$\times$2 = 256
  \item Masked global avg pool $\to$ FC(256$\to$256)
  \item 7 FiLM generators ($\gamma$, $\beta$ per stage)
\end{itemize}

\medskip
\textbf{FiLM Conditioning:} $y = \gamma \cdot x + \beta$\\
Channel-wise affine transform; zero spatial cost.
Identity init ($\gamma$$\to$1, $\beta$$\to$0).

\end{columns}
\end{frame}

% SLIDE 7 — LOSS FUNCTION
% ═══════════════════════════════════════════════════════════════════
\begin{frame}{Loss Function \& Optimiser}
\scriptsize

\textbf{Compound V2 Loss:}
\vspace{-0.5em}
\[
\mathcal{L}_{\text{total}} = \underbrace{1.0 \cdot \mathcal{L}_{\text{FT}}}_{\text{Focal Tversky}} \;+\; \underbrace{0.5 \cdot \mathcal{L}_{\text{Boundary}}}_{\text{Edge Focus}} \;+\; \underbrace{0.5 \cdot \mathcal{L}_{\text{OHEM}}}_{\text{Hard Pixels}}
\]
\vspace{-0.5em}

\begin{columns}[T]
\column{0.32\textwidth}
\begin{block}{Focal Tversky}
$\mathcal{L}_{\text{FT}} = (1 - TI)^\gamma$\\[0.1em]
$TI = \frac{TP}{TP + 0.7{\cdot}FN + 0.3{\cdot}FP}$\\[0.1em]
$\gamma = 0.75$ \\[0.1em]
\textit{Heavy FN penalty --- missing a crack is worse than a false alarm.}
\end{block}

\column{0.32\textwidth}
\begin{block}{Boundary Loss}
Edge = dilate $-$ erode (3$\times$3)\\[0.1em]
5$\times$ weight on boundary pixels\\[0.1em]
\textit{Sharpens crack edges instead of blobby masks.}
\end{block}

\column{0.32\textwidth}
\begin{block}{OHEM Loss}
Top 25\% hardest pixels (BCE)\\[0.1em]
\textit{Cracks are $<$2\% of pixels; OHEM prevents ignoring them.}
\end{block}
\end{columns}

\medskip
\textbf{Deep Supervision:}
$\mathcal{L}_{\text{deep}} = \mathcal{L}_{\text{base}}(\text{main}) + 0.4 \cdot \mathcal{L}_{\text{base}}(\text{aux}_1) + 0.2 \cdot \mathcal{L}_{\text{base}}(\text{aux}_2)$
\; (aux at $\frac{1}{32}$, $\frac{1}{16}$ res)

\smallskip
\textbf{Optimiser:} Adam ($\beta_1$=0.9, $\beta_2$=0.999)
\;|\; LR: 5$\times$10$^{-4}$
\;|\; Cosine + 10-ep warmup
\;|\; WD: 10$^{-4}$
\;|\; Clip: 5.0
\;|\; AMP: Yes

\end{frame}

\begin{frame}{Conclusion}
\footnotesize

\begin{columns}[T]
\column{0.55\textwidth}

\textbf{What we built:}
\begin{itemize}
  \item Text-prompted segmentation model trained \emph{from scratch} --- no pretrained weights, no transfer learning
  \item Single lightweight model handles two damage classes via FiLM conditioning from a character-level BiLSTM
  \item Iterative V1$\to$V2 design: higher resolution (256$\to$512), ASPP for multi-scale context, Focal Tversky + Boundary + OHEM loss, deep supervision
  \item Final: \textbf{mIoU 0.73}, \textbf{Dice 0.78}, \textbf{103\,FPS}, \textbf{2.75\,M params}
\end{itemize}

\medskip
\textbf{Remaining bottleneck:}
\begin{itemize}
  \item Crack data quality, not model capacity
  \item Very thin cracks ($<$1\% foreground) achieve IoU $\sim$0.44
  \item Industry path: better data annotation $>$ bigger model
\end{itemize}

\column{0.42\textwidth}

  \item See mask overlay + probability heatmap

\end{itemize}
